        // Crear part√≠culas flotantes (responsive)
        function createParticles() {
            const particlesContainer = document.getElementById('particles');
            
            // Ajustar n√∫mero de part√≠culas seg√∫n el tama√±o de pantalla
            let particleCount = 50;
            if (window.innerWidth < 768) {
                particleCount = 20; // Menos part√≠culas en m√≥vil para mejor rendimiento
            } else if (window.innerWidth < 1200) {
                particleCount = 35; // Cantidad media en tablet
            }

            for (let i = 0; i < particleCount; i++) {
                const particle = document.createElement('div');
                particle.className = 'particle';
                
                // Posici√≥n aleatoria
                particle.style.left = Math.random() * 100 + '%';
                particle.style.top = Math.random() * 100 + '%';
                
                // Delay de animaci√≥n aleatorio
                particle.style.animationDelay = Math.random() * 6 + 's';
                particle.style.animationDuration = (Math.random() * 3 + 4) + 's';
                
                particlesContainer.appendChild(particle);
            }
        }

        // Countdown timer - FECHA FIJA: 7 de Julio 2025
        function startCountdown() {
            // Fecha objetivo fija: 7 de Julio 2025 a las 00:00:00 (hora de Colombia)
            const targetDate = new Date('July 7, 2025 00:00:00 GMT-0500');
            
            function updateCountdown() {
                const now = new Date().getTime();
                const distance = targetDate.getTime() - now;

                // Si ya pas√≥ la fecha objetivo
                if (distance < 0) {
                    clearInterval(countdownInterval);
                    document.getElementById('countdown').innerHTML = 
                        '<div class="launch-message">üéâ ¬°Ya estamos aqu√≠! üéâ</div>';
                    return;
                }

                // Calcular tiempo restante
                const days = Math.floor(distance / (1000 * 60 * 60 * 24));
                const hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
                const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
                const seconds = Math.floor((distance % (1000 * 60)) / 1000);

                // Actualizar display
                document.getElementById('days').textContent = days.toString().padStart(2, '0');
                document.getElementById('hours').textContent = hours.toString().padStart(2, '0');
                document.getElementById('minutes').textContent = minutes.toString().padStart(2, '0');
                document.getElementById('seconds').textContent = seconds.toString().padStart(2, '0');
            }

            // Actualizar cada segundo
            const countdownInterval = setInterval(updateCountdown, 1000);
            updateCountdown(); // Llamada inicial
        }

        // Inicializar todo cuando la p√°gina cargue
        window.addEventListener('load', function() {
            createParticles();
            startCountdown();
            
            // Animaci√≥n de entrada
            document.querySelector('.content').style.opacity = '0';
            document.querySelector('.content').style.transform = 'translateY(50px)';
            
            setTimeout(() => {
                document.querySelector('.content').style.transition = 'all 1s ease';
                document.querySelector('.content').style.opacity = '1';
                document.querySelector('.content').style.transform = 'translateY(0)';
            }, 300);
        });

        // Recrear part√≠culas cuando cambie el tama√±o de ventana
        let resizeTimeout;
        window.addEventListener('resize', function() {
            clearTimeout(resizeTimeout);
            resizeTimeout = setTimeout(() => {
                const particlesContainer = document.getElementById('particles');
                particlesContainer.innerHTML = ''; // Limpiar part√≠culas existentes
                createParticles(); // Crear nuevas con el n√∫mero apropiado
            }, 250);
        });

        // Parallax suave en scroll (optimizado)
        let ticking = false;
        window.addEventListener('scroll', function() {
            if (!ticking) {
                requestAnimationFrame(function() {
                    const scrolled = window.pageYOffset;
                    const particles = document.getElementById('particles');
                    if (particles) {
                        particles.style.transform = `translateY(${scrolled * 0.3}px)`;
                    }
                    ticking = false;
                });
                ticking = true;
            }
        });
